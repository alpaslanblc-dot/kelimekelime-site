Mevcut Kelimeler

${k.kelime}
`; }); } } // 4. Silme Fonksiyonu async function sil(id) { if(confirm("Emin misin?")) { await _supabase.from('kelimeler').delete().eq('id', id); location.reload(); } } // Sayfa açıldığında listeyi yükle listele();
